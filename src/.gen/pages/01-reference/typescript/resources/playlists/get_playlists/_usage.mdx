import { PlexAPI } from "@lukehagar/plexjs";
import { PlaylistType, QueryParamSmart } from "@lukehagar/plexjs/models/operations";

  });

  const playlistType = PlaylistType.Video;
  const smart = QueryParamSmart.Zero;

  console.log(result)
}

run();
